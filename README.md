libffi-3.2.1-ia64-11.31.depot<br>
libiconv-1.15-ia64-11.31.depot<br>
libunistring-0.9.7-ia64-11.31.depot<br>

openssl-1.0.2k-ia64-11.31.depot<br>
python-2.7.6-ia64-11.31.depot<br>
